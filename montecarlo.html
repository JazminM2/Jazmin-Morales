### fitt of the data

decay <- function(x, params) {
    a <- params[1]
    b <- params[2]
    d <- a*x**b
    return(d)
}

## function to minimize
fun.to.minimize <- function(params,x,y,var){
        a <- params[1]
        b <- params[2]
        flost <- a*x^b
        output <- sum((y-flost)^2/var^2)
        return(output)
    }

## initial conditions

a0 <- 2
b0 <- -2

## fitting
solution <- optim(c(a0,b0), fun.to.minimize, x=xMeas, y=yMeas, var=sError)
new.a <- solution$par[1]
new.b <- solution$par[2]

trial <- 10
m <- 2000  ## number of iterations

par1 <- c()  ## parameter 1
par2 <- c()  ## parameter 2

for (i in 1:m){
    xTrial = runif(trial, 0.5,3.0);
    yGen = 1.5/xTrial;
    yTrial = yGen + rnorm(trial, yGen, sError);
    fit <- optim(c(a0,b0), fun.to.minimize, x=xTrial, y=yTrial, var=sError);
    #lines(xTrial, decay(xTrial, c(fit$par[1],fit$par[2])), col='green')
    par1 <- append(par1, fit$par[1])
    par2 <- append(par2, fit$par[2])
}

b) Application to error propagation to Newton's Law [2]:

$$ F = G \frac{M_1 M_2}{r^2} $$

Given the following values and uncertainties:

$M_1 = 40 \pm 0.05 $ kg

$M_2 = 30 \pm 0.1 $ kg

$r = 5.3 \pm 0.001 $ m

$G$ is considered as an exact value:

$G = 6.674 \times 10^{-11}$ m$^3$/kgs$^2$

In [11]:
## N iterations
N = 10000

## values and uncertainties

G <- 6.674E-11

m1 <- rnorm(N, 4000, 0.5)
m2 <- rnorm(N, 3000, 0.1)
r <- rnorm(N, 53, 0.5)


## force F

force <- function(m1, m2, r){
    f = G*m1*m2/r^2 
    return(f)
}



## calculating the force

F <- force(m1, m2, r)
